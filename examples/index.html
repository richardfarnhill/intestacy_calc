Quick Start

Get up and running in under a minute. Choose npm for build-tool projects or a script tag for instant integration.

Script tag
<!-- Add a container element -->
<div id="calculator"></div>

<!-- Include the script -->
<script src="intestacy-calculator.js"></script>
<script>
  new IntestacyCalculator({
    container: '#calculator',
    theme: 'light'
  });
</script>
